338  !
339  ! We need to convert the pointer values we receive from the C
340  ! library to equivalent Fortran procedure pointers in order
341  ! to use them. We can do this with the intrinsic c_f_procpointer

347  !
348  ! Similarly, when we passed in the user_arg pointer we used
349  ! the c_loc function. To convert this value back into a
350  ! Fortran pointer to our data, we undo this with the
351  ! intrinsic c_f_pointer subroutine.
352  !
353  call c_f_pointer(user_arg, data_pointer)

355  !
356  ! The actual forward model code
357  !
358  sum = 0.0
359  do i = 1, datasize
360 
361  !
362  ! We can use the value_at procedure pointer to query the proposed
363  ! velocity profile versus time at different times. This method
364  ! returns the number of local parameters (1 in this example) and
365  ! fills the values array with the local parameter values at the
366  ! point data_pointer%t(i). After this call is successfull, the
367  ! proposed velocity at the given time will be the first element
368  ! of the array values.
369  !
370  j = value_at(state, data_pointer%x(i), nparameters, values)
371 
372  py = values(1)
373 
374  !
375  ! Accumulate the sum of the square errors between the displacement
376  ! profile derived from the proposed velocity profile and the
377  ! measured displacement.
378  !
379  sum = sum + (py - data_pointer%sy(i))**2
380 
381  end do
382

384  ! Return the sum of the square errors.
385  !
386 
387  !write (*,*) sum
